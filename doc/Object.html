def inspect_or_to_s
  return 'empty'            if self == ''
  return self.to_s          if self.class == String
  return self.inspect_see   if self.class == Time
  return self.inspect_see   if self.kind_of?(Numeric)
  return self.inspect_see   if self.kind_of?(Set)
  return self.inspect
end

def inspect_see( wrap = 200 )
  result = ''
  result = PPP::pp(self, result, wrap)  # normal prettyprint
  result = PPP.beautify(result)
  return result
end

def log(input=:kein_input, *rest)
  out_bak = seee.out.dup
  seee.out=[:log]
  if rest.empty?
    see input
  else
    seee.process_print( ( [input] + rest), :method => :multi )
  end
  seee.out=out_bak
  nil
end

def see(input=:kein_input, *rest)

  if input == :kein_input  
    seee.process_print( "\n", :method => :puts )
    return ''
  end        
  
  return seee.process_print( ( [input] + rest), :method => :multi )    unless rest.empty?
  return see_puts(input)              if input.class == String
  return see_pp(input)                if input.class == Hash
  return see_puts(input)              unless (input.respond_to?(:compact)  ||  input.instance_variables )   # Ordin√§re Objekte macht puts
  return see_pp(input)        # Den Rest macht pp    
                          
end

def see_flash(input=:kein_input, alternativtext=' ', times=6)
  if input == :kein_input  
    seee.delayed_newlines += 1

  end
